<script type="text/javascript">

function geocode(){

	var where = prompt('where');

	if (! where){
		return;
	}

	var api_key = '';

	var url = 'http://api.flickr.com/services/rest/?method=flickr.places.find&format=json&nojsoncallback=1';
	url += '&api_key=' + api_key;
	url += '&query=' + where;

	$.ajax({
		'url' : url,
		'success' : function(rsp){

			if (rsp['stat'] != 'ok'){
				return;
			}

			// do stuff here...
		}
	});
}

</script>
